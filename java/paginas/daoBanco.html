DAO com MySQL

Neste exemplo utilizamos o padrão DAO (Data Access Object) com um banco de dados MySQL, permitindo a persistência real dos dados.

Dependência (Driver JDBC)



    mysql
    mysql-connector-java
    8.0.33

Classe de Conexão


import java.sql.Connection;
import java.sql.DriverManager;

public class Conexao {

    public static Connection conectar() throws Exception {
        String url = "jdbc:mysql://localhost:3306/teste";
        String user = "root";
        String password = "";

        return DriverManager.getConnection(url, user, password);
    }
}

Classe DAO (CRUD Completo)


import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;

public class PessoaDAO {

    // CREATE
    public void inserir(Pessoa p) throws Exception {

        Connection conn = Conexao.conectar();

        String sql = "INSERT INTO pessoa (nome, idade) VALUES (?, ?)";

        PreparedStatement stmt = conn.prepareStatement(sql);
        stmt.setString(1, p.getNome());
        stmt.setInt(2, p.getIdade());

        stmt.execute();
        conn.close();
    }

    // READ (listar todos)
    public void listar() throws Exception {

        Connection conn = Conexao.conectar();

        String sql = "SELECT * FROM pessoa";

        PreparedStatement stmt = conn.prepareStatement(sql);
        ResultSet rs = stmt.executeQuery();

        while(rs.next()){
            System.out.println(rs.getInt("id") + " - " +
                               rs.getString("nome") + " - " +
                               rs.getInt("idade"));
        }

        conn.close();
    }

    // READ (buscar um)
    public Pessoa buscarUm(int id) throws Exception {

        Connection conn = Conexao.conectar();

        String sql = "SELECT * FROM pessoa WHERE id = ?";

        PreparedStatement stmt = conn.prepareStatement(sql);
        stmt.setInt(1, id);

        ResultSet rs = stmt.executeQuery();

        Pessoa pessoa = null;

        if(rs.next()){
            pessoa = new Pessoa(
                rs.getString("nome"),
                rs.getInt("idade")
            );
        }

        conn.close();

        return pessoa;
    }

    // UPDATE
    public void atualizar(int id, Pessoa p) throws Exception {

        Connection conn = Conexao.conectar();

        String sql = "UPDATE pessoa SET nome = ?, idade = ? WHERE id = ?";

        PreparedStatement stmt = conn.prepareStatement(sql);
        stmt.setString(1, p.getNome());
        stmt.setInt(2, p.getIdade());
        stmt.setInt(3, id);

        stmt.execute();

        conn.close();
    }

    // DELETE
    public void excluir(int id) throws Exception {

        Connection conn = Conexao.conectar();

        String sql = "DELETE FROM pessoa WHERE id = ?";

        PreparedStatement stmt = conn.prepareStatement(sql);
        stmt.setInt(1, id);

        stmt.execute();

        conn.close();
    }
}

Estrutura da Tabela


CREATE TABLE pessoa (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    idade INT
);

Explicação

  • Connection → responsável pela conexão com o banco
  • PreparedStatement → executa comandos SQL com segurança
  • ResultSet → armazena o resultado das consultas
  • id → identifica cada registro (chave primária)

Resumo

  • DAO separa a lógica de acesso aos dados
  • MySQL permite persistência real
  • CRUD completo: inserir, listar, buscar, atualizar e excluir

Sistema Completo (MVC + DAO + MySQL)

Nesta etapa vamos construir um sistema completo utilizando Model, View, Controller, DAO e Banco de Dados.

Agora não estamos mais trabalhando com exemplos isolados, mas com uma estrutura semelhante à utilizada no mercado.

Organização em Pacotes

Antes de iniciar, organize seu projeto criando os seguintes pacotes: 


model
view
controller
dao

Essa separação facilita a organização e manutenção do sistema.

Fluxo da aplicação


Main → View → Controller → DAO → Banco de Dados

Cada camada possui uma responsabilidade específica.

  • View → interação com o usuário
  • Controller → controla o fluxo
  • DAO → acessa os dados
  • Banco → armazena os dados

Model (Pessoa)


package model;

public class Pessoa {

    private String nome;
    private int idade;

    public Pessoa(String nome, int idade){
        this.nome = nome;
        this.idade = idade;
    }

    public String getNome() {
        return nome;
    }

    public int getIdade() {
        return idade;
    }
}

Conexão com o Banco


package dao;

import java.sql.Connection;
import java.sql.DriverManager;

public class Conexao {

    public static Connection conectar() throws Exception {
        String url = "jdbc:mysql://localhost:3306/teste";
        String user = "root";
        String password = "";

        return DriverManager.getConnection(url, user, password);
    }
}

DAO (MySQL)

O DAO é responsável por executar comandos SQL no banco de dados. 


package dao;

import model.Pessoa;
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;

public class PessoaDAO {

    public void inserir(Pessoa p) throws Exception {

        Connection conn = Conexao.conectar();

        String sql = "INSERT INTO pessoa (nome, idade) VALUES (?, ?)";

        PreparedStatement stmt = conn.prepareStatement(sql);
        stmt.setString(1, p.getNome());
        stmt.setInt(2, p.getIdade());

        stmt.execute();
        conn.close();
    }

    public void listar() throws Exception {

        Connection conn = Conexao.conectar();

        String sql = "SELECT * FROM pessoa";

        PreparedStatement stmt = conn.prepareStatement(sql);
        ResultSet rs = stmt.executeQuery();

        while(rs.next()){
            System.out.println(rs.getInt("id") + " - " +
                               rs.getString("nome") + " - " +
                               rs.getInt("idade"));
        }

        conn.close();
    }

    public Pessoa buscarUm(int id) throws Exception {

        Connection conn = Conexao.conectar();

        String sql = "SELECT * FROM pessoa WHERE id = ?";

        PreparedStatement stmt = conn.prepareStatement(sql);
        stmt.setInt(1, id);

        ResultSet rs = stmt.executeQuery();

        Pessoa p = null;

        if(rs.next()){
            p = new Pessoa(
                rs.getString("nome"),
                rs.getInt("idade")
            );
        }

        conn.close();
        return p;
    }

    public void atualizar(int id, Pessoa p) throws Exception {

        Connection conn = Conexao.conectar();

        String sql = "UPDATE pessoa SET nome = ?, idade = ? WHERE id = ?";

        PreparedStatement stmt = conn.prepareStatement(sql);
        stmt.setString(1, p.getNome());
        stmt.setInt(2, p.getIdade());
        stmt.setInt(3, id);

        stmt.execute();
        conn.close();
    }

    public void excluir(int id) throws Exception {

        Connection conn = Conexao.conectar();

        String sql = "DELETE FROM pessoa WHERE id = ?";

        PreparedStatement stmt = conn.prepareStatement(sql);
        stmt.setInt(1, id);

        stmt.execute();
        conn.close();
    }
}

Controller

O Controller recebe dados da View e chama o DAO. 


package controller;

import dao.PessoaDAO;
import model.Pessoa;

public class PessoaController {

    private PessoaDAO dao = new PessoaDAO();

    public void adicionar(String nome, int idade) throws Exception {
        dao.inserir(new Pessoa(nome, idade));
    }

    public void listar() throws Exception {
        dao.listar();
    }

    public Pessoa buscar(int id) throws Exception {
        return dao.buscarUm(id);
    }

    public void atualizar(int id, String nome, int idade) throws Exception {
        dao.atualizar(id, new Pessoa(nome, idade));
    }

    public void excluir(int id) throws Exception {
        dao.excluir(id);
    }
}

View

Responsável pela interação com o usuário. 


package view;

import java.util.Scanner;

public class PessoaView {

    private Scanner sc = new Scanner(System.in);

    public int menu(){
        System.out.println("1 - Adicionar");
        System.out.println("2 - Listar");
        System.out.println("3 - Buscar");
        System.out.println("4 - Atualizar");
        System.out.println("5 - Excluir");
        System.out.println("0 - Sair");
        return sc.nextInt();
    }

    public String lerNome(){
        System.out.print("Nome: ");
        return sc.next();
    }

    public int lerIdade(){
        System.out.print("Idade: ");
        return sc.nextInt();
    }

    public int lerId(){
        System.out.print("ID: ");
        return sc.nextInt();
    }

    public void msg(String texto){
        System.out.println(texto);
    }
}

Main

Integra todas as partes do sistema. 


import controller.PessoaController;
import view.PessoaView;
import model.Pessoa;

public class Main {

    public static void main(String[] args) throws Exception {

        PessoaController controller = new PessoaController();
        PessoaView view = new PessoaView();

        int op;

        do {
            op = view.menu();

            switch(op){

                case 1:
                    controller.adicionar(view.lerNome(), view.lerIdade());
                    view.msg("Salvo!");
                    break;

                case 2:
                    controller.listar();
                    break;

                case 3:
                    Pessoa p = controller.buscar(view.lerId());
                    if(p != null){
                        view.msg(p.getNome());
                    } else {
                        view.msg("Não encontrado");
                    }
                    break;

                case 4:
                    int id = view.lerId();
                    controller.atualizar(id, view.lerNome(), view.lerIdade());
                    view.msg("Atualizado!");
                    break;

                case 5:
                    controller.excluir(view.lerId());
                    view.msg("Excluído!");
                    break;
            }

        } while(op != 0);
    }
}
